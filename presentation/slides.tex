\begin{frame}[fragile]       % Frame 1 title
\frametitle{Vanilla \LaTeX / beamer stuff}
\note{Enabling/disabling notes:}
\note[item]{Show only slides: \texttt{\textbackslash setbeameroption\{hide notes\}}}
\note[item]{Show only notes: \texttt{\textbackslash setbeameroption\{show only notes\}}}
\note[item]{Show both (for use with e.g.~\href{https://github.com/Cimbali/pympress}{Pympress} during actual presentations) \texttt{\textbackslash setbeameroption\{show notes on second screen=right\}}}

\begin{itemize}
    \item \emph{\textbackslash emph\{\}}, \alert{\textbackslash alert\{\}}
    \item Citation: \citet{touvron2023llama}, \citep{touvron2023llama}, \footcite{touvron2023llama}
    \item Verbatim text
        \begin{itemize}
            \item Needs \texttt{\textbackslash begin\{frame\}[fragile]}
            \item Inline: \texttt{\textbackslash verb$\vert$text$\vert$} (less
            escaping than \verb|\texttt{}|)
            \item Block text: \texttt{\textbackslash begin\{verbatim\}}
        \end{itemize}
    \item Emoji
        \begin{itemize}
            \item \texttt{\textbackslash usepackage\{fontspec\}} (already in
                \texttt{header.tex})\\
            \texttt{\textbackslash usepackage\{emoji\}}(already in
                \texttt{header.tex})\\
            \texttt{\textbackslash emoji\{hugging-face\}} $\Rightarrow$ \emoji{hugging-face}
        \end{itemize}
\end{itemize}
\end{frame}
